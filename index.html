Bar Plot of Scenarios (2000–2050)

import matplotlib.pyplot as plt import numpy as np # Years: 2000–2050 at 5-year intervals years = np.arange(2000, 2055, 5) # Create 5 scenarios (random or simulated data) np.random.seed(42) scenarios = { f"Scenario {i+1}": np.random.randint(50, 200, len(years)) for i in range(5) } # Plot as grouped bar chart bar_width = 0.15 x = np.arange(len(years)) fig, ax = plt.subplots(figsize=(12, 6)) for i, (name, values) in enumerate(scenarios.items()): ax.bar(x + i*bar_width, values, bar_width, label=name) ax.set_xticks(x + bar_width*2) ax.set_xticklabels(years, rotation=45) ax.set_xlabel("Year") ax.set_ylabel("Value") ax.set_title("Scenario Comparison (2000–2050)") ax.legend() plt.tight_layout() plt.show()
